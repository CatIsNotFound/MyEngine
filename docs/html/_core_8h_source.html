219 static void log(const std::string& message, LogLevel level = DEBUG) {
220 _running_time = SDL_GetTicksNS();
221 if (level < _base_level) return;
222 auto _real_time = (_running_time - _started_time) / 1e9;
223 if (level >= WARN) {
224 std::cerr << std::format("[{:.06f}] [{}] {}\n", _real_time, _logLevelToString(level), message);
225 } else {
226 std::cout << std::format("[{:.06f}] [{}] {}\n", _real_time, _logLevelToString(level), message);
227 }
228 _last_log_level = level;
229 _last_log_info = message;
230 }
231
232 static void setBaseLogLevel(LogLevel level) {
233 _base_level = level;
234 }
235
236 static std::string lastError() {
237 return (_last_log_level >= ERROR ? _last_log_info : "");
238 }
239 private:
240 static std::string _logLevelToString(LogLevel level) {
241 switch (level) {
242 case DEBUG: return "DEBUG";
243 case INFO: return "INFO";
244 case WARN: return "WARN";
245 case ERROR: return "ERROR";
246 case FATAL: return "FATAL";
247 default: return "UNKNOWN";
248 }
249 }
