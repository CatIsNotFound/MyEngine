184 void setCursor(const StdCursor& cursor) {
185 if (this->cursor() == cursor) return;
186 if (_custom_cursor && _user_custom.contains(cursor)) {
187 auto user_cursor = _user_custom.at(cursor);
188 if (user_cursor.cursor) {
189 SDL_DestroyCursor(_cursor);
190 }
191 _cursor = SDL_CreateColorCursor(user_cursor.cursor.get(),
192 user_cursor.hot_point.x, user_cursor.hot_point.y);
193 SDL_SetCursor(_cursor);
194 _std_cursor = cursor;
195 return;
196 }
197 _std_cursor = cursor;
198 SDL_DestroyCursor(_cursor);
199 if (_surface) {
200 SDL_DestroySurface(_surface);
201 _surface = nullptr;
202 }
203 _cursor = SDL_CreateSystemCursor(SStdCursor(_std_cursor));
204 SDL_SetCursor(_cursor);
205 }

252 void setCustomCursor(const StdCursor& stdCursor, const std::string& path, const UserCustom::HotPoint& hot_point) {
253 auto new_cursor = IMG_Load(path.c_str());
254 if (!new_cursor) {
255 SDL_Log("[ERROR] Can't set custom cursor, because the current path \"%s\" is not valid!", path.c_str());
256 return;
257 }
258 if (_user_custom.contains(stdCursor)) {
259 if (_user_custom.at(stdCursor).cursor) SDL_DestroySurface(_user_custom.at(stdCursor).cursor.get());
260 _user_custom.at(stdCursor).cursor = std::unique_ptr<SDL_Surface>(new_cursor);
261 _user_custom.at(stdCursor).hot_point = hot_point;
262 } else {
263 _user_custom.emplace(stdCursor, UserCustom(hot_point, new_cursor));
264 }
265 }
